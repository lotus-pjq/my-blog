```C++
//欧拉函数+莫比乌斯函数预处理
const int N=2e5+5;
const int M=2e6;
bool vis[M];
int p[N],mu[M],phi[M];//莫比乌斯函数和欧拉函数
int mu_pre[M],phi_pre[N];//前缀和
void init(){
  mu[1]=phi[1]=1;
  for(int i=2;i<M;i++){
    if(!vis[i]) p[++p[0]]=i,mu[i]=-1,phi[i]=i-1;//φ(p)=p-1
    for(int j=1;j<=p[0]&&i*p[j]<M;j++){
      vis[i*p[j]]=1;
      if(i%p[j]==0) break;
      mu[i*p[j]]=-mu[i];
      phi[i*p[j]]=phi[i]*(p[j]-1);
    }
  }
  for(int i=1;i<M;i++) mu_pre[i]=mu_pre[i-1]+mu[i];
  for(int i=1;i<M;i++) phi_pre[i]=phi_pre[i-1]+phi[i];
}
```
