- 1007.性质不同的数字
  异或哈希
  ```C++
  mt19937_64 rg(random_device{}());
  struct node{
  	int x,tag;
  	bool operator<(const node&tmp)const{return x>tmp.x;}
  };
  priority_queue<node> q;
  unordered_map<int,bool> vis;
  void solve(){
  	vis.clear();
  	int n,l,r;cin>>n;
  	for(int i=1;i<=n;i++){
  		cin>>l>>r;
  		int ran=rg();
  		q.push({l,ran});
  		q.push({r+1,ran});
  	}
  	vis[0]=1;
  	int cur=0,ans=1;
  	while(!q.empty()){
  		int x=q.top().x,tag=q.top().tag;q.pop();
  		cur^=tag;
  		if(!q.empty()&&q.top().x==x) continue;
  		if(!vis[cur]) ans++,vis[cur]=1;
  	}
  	cout<<ans<<endl;
  }
  ```
