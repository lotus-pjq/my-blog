## 1009乘法逆元
	- #异或
	- 题目：
	  
		- 代码：
		  ```C++
		  #include<bits/stdc++.h>
		  using namespace std;
		  #define int long long
		  #define endl '\n'
		  const int mod=998244353;
		  const int N=2e5+5;
		  int h,n,m,a[N];
		  __int128_t ans,p;
		  __int128_t ksm(int x,__int128_t y,int mod=998244353){__int128_t ans=1;while(y){if(y&1)ans=ans*x%mod;x=x*x%mod;y>>=1;}return ans;}
		  __int128_t ksm2(int x,__int128_t y){__int128_t ans=1;while(y){if(y&1)ans=ans*x;x=x*x;y>>=1;}return ans;}
		  void solve(){
		  	cin>>h;p=h;
		  	int k=(p-1)/119+1;
		  	ans=0;
		  	if(p<=3000){
		  		for(int i=1;i<p;i++){
		  			__int128_t a=ksm(i,p-2,p)+ksm2(2,k);
		  			__int128_t b=i+ksm2(4,k);
		  			ans^=a*b;
		  		}
		  	}
		  	else{
		  		int b;
		  		if(p%4==1) b=(p-1)%mod;
		  		else b=p%mod;
		  		__int128_t pw=ksm(2,k);
		  		ans=b*(pw+pw*pw%mod)%mod;
		  		ans+=(1ll^((p-1)*(p-1)));
		  	}
		  	ans%=mod;int g=ans;
		  	cout<<g<<endl;
		  }
		  signed main(){
		  	ios::sync_with_stdio(false);
		  	cin.tie(nullptr);
		  	int t;cin>>t;
		  	while(t--) solve();
		  	return 0;
		  }

- ## 1005计算几何

	- ```C++
	  #include<bits/stdc++.h>
	  using namespace std;
	  #define rep(i,j,k) for (int i=j;i<=k;++i)
	  #define int long long
	  #define endl '\n'
	  //int up=*max_element(a+1,a+n+1);

	  const int N=5e5+5;
	  struct point{
	      int x,y;
	      point(int x, int y):x(x),y(y){}
	      point():x(0),y(0){}
	  }p[N];
	  struct line{point s,e;};
	  point operator+(point a,point b){return point(a.x+b.x,a.y+b.y);}
	  point operator-(point a,point b){return point(a.x-b.x,a.y-b.y);}
	  point operator*(point a,int t){return point(a.x*t,a.y*t);}
	  int operator*(point a,point b){return a.x*b.y-a.y*b.x;}//叉积
	  int operator&(point a,point b){return a.x*b.x+a.y*b.y;}//点积
	  //点排序
	  //点去共线（差积=0）
	  //左转则NO,右转是YES
	  int n;
	  int x[N];
	  int y[N];
	  struct node{
	  	int x,y;
	  	int jud;
	  }u[N];
	  int ans[N];
	  bool cmp(node a,node b){return a.x!=b.x?a.x<b.x:a.y<b.y;}
	  void solve(){
	      cin>>n;int i;
	      for(i=1;i<=n;i++){
	          cin>>x[i]>>y[i];
	      }int mu=1;
	      for(i=2;i<=n;i++){
	          if(x[i]<x[mu]) mu=i;
	      	else if(x[i]==x[mu]&&y[i]<y[mu]) mu=i;
	      }
	  	int z=0;
	      for(i=mu;i<=n;i++) p[++z]={x[i],y[i]};
	      for(i=1;i<mu;i++) p[++z]={x[i],y[i]};
	  	int num=0;
	  	int cal=(p[2]-p[1])*(p[1]-p[z]);
	  	if(cal>0) u[++num]={p[1].x,p[1].y,1};
	  	else if(cal<0) u[++num]={p[1].x,p[1].y,-1};
	  
	  	for(int i=2;i<=z-1;i++){
	  		cal=(p[i+1]-p[i])*(p[i]-p[i-1]);
	  		if(cal==0) continue;
	  		else if(cal>0) u[++num]={p[i].x,p[i].y,1};
	  		else u[++num]={p[i].x,p[i].y,-1};
	  	}
	  
	  	cal=(p[1]-p[z])*(p[z]-p[z-1]);
	  	if(cal>0) u[++num]={p[z].x,p[z].y,1};
	  	else if(cal<0) u[++num]={p[z].x,p[z].y,-1};
	  
	  	cout<<num<<endl;
	  	if(u[1].jud==-1){
	  		for(int i=1;i<=num;i++)
	  			u[i].jud=-u[i].jud;
	  	}
	  	sort(u+1,u+num+1,cmp);
	  	for(int i=1;i<=num;i++){
	  		cout<<u[i].x<<" "<<u[i].y<<" ";
	  		if(u[i].jud==1) cout<<"YES"<<endl;
	  		else cout<<"NO"<<endl;
	  	}
	  }
	  signed main(){
	  	ios::sync_with_stdio(false);
	  	cin.tie(nullptr);
	  	int t;cin>>t;
	  	while(t--) solve();
	  	return 0;
	  }
