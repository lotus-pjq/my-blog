## P4151 [WC2011] 最大XOR和路径
- #图论 #线性基
- - 
- 思路：- 我们枚举所有环，将环上异或和扔进线性基，然后用这条链作为初值，求线性基与这条链的最大异或和。
  - 怎么选最开始的这条链，其实它可以随便选
- Code:
```C++
#include
using namespace std;
#define int long long
#define endl '\n'
#define rep(i,j,k) for (int i=j;i g[N];
int cal[N];
int p[64];
void insert(int x){//贪心法
  for(int i=63;i>=0;--i)
    if(x>>i&1){
      if(!p[i]){p[i]=x;break;}//不存在则加入
      x^=p[i];//存在则异或
    }
}
void dfs(int u,node fa){
  for(node v:g[u]){
    if(v==fa) continue;
    if(cal[v.id]!=-1) insert(cal[u]^cal[v.id]^v.val);
    else{
      cal[v.id]=cal[u]^v.val;
      dfs(v.id,{u,v.val});
    }
  }
}
signed main(){
  ios::sync_with_stdio(false);
  cin.tie(nullptr);
  memset(cal,-1,sizeof(cal));
  cin>>n>>m;
  int u,v,w;
  rep(i,1,m){
    cin>>u>>v>>w; 
    g[u].pb({v,w}),g[v].pb({u,w});
  }
  cal[1]=0;
  dfs(1,{-1,-1});
  int ans=cal[n];
  //cout=0;i--) ans=max(ans,ans^p[i]);
  cout<<ans;
}
```
