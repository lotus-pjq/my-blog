- ```C++
  int digitCnt(int n){
  	if(n<=0) return 0;
  	int sum=0;
  	int cur=1;//当前位权
  	int digit=1;//当前数字位数
  	while(cur<=n){
  		int nxt=cur*10;
  		int cnt=min(n,nxt-1)-cur+1;
  		sum+=cnt*digit;
  		cur=nxt;
  		digit++;
  	}
  	return sum;
  }
  ```
