- 基本的分块操作（记录块的信息）
  ```C++
  int n;cin>>n;
  int B=sqrt(n);
  int t=n/B + (n%B);
  for(int i=1;i<=t;i++)
    st[i]=(i-1)*B+1, ed[i]=i*B;
  ed[t]=n;
  for(int i=1;i<=n;i++)
    pos[i]=(i-1)/B+1;
  ```
- 区间修改+区间查询板子题（luogu P3372)
  ```C++
  const int N=1e5+5;
  int st[N],ed[N],pos[N];
  int a[N],sum[N],add[N];
  void change(int l,int r,int d){
  	int p=pos[l],q=pos[r];//p:起始块，q:末尾块
  	if(p==q){//没有整块
  		for(int i=l;i<=r;i++) a[i]+=d;
  		sum[p]+=d*(r-l+1);
  	}else{
  		for(int i=p+1;i<=q-1;i++) add[i]+=d;
  		for(int i=l;i<=ed[p];i++) a[i]+=d;//l->ed[p]
  		sum[p]+=d*(ed[p]-l+1);
  		for(int i=st[q];i<=r;i++) a[i]+=d;//st[q]->r
  		sum[q]+=d*(r-st[q]+1);
  	}
  }
  int query(int l,int r){
  	int p=pos[l],q=pos[r], ans=0;
  	if(p==q){
  		for(int i=l;i<=r;i++) ans+=a[i];
  		ans+=add[p]*(r-l+1);
  	}else{
  		for(int i=p+1;i<=q-1;i++) ans+=sum[i]+add[i]*(ed[i]-st[i]+1);
  		for(int i=l;i<=ed[p];i++) ans+=a[i];
  		ans+=(ed[p]-l+1)*add[p];
  		for(int i=st[q];i<=r;i++) ans+=a[i];
  		ans+=(r-st[q]+1)*add[q];
  	}
  	return ans;
  }
  void init(int n){for(int i=1;i<=n;i++) sum[pos[i]]+=a[i];}
  signed main(){
  	ios::sync_with_stdio(false);
  	cin.tie(nullptr);
  	int n,q; cin>>n>>q;
  	for(int i=1;i<=n;i++) cin>>a[i];
  	int B=sqrt(n);
  	int t=n/B + (n%B);
  	for(int i=1;i<=t;i++)
  		st[i]=(i-1)*B+1, ed[i]=i*B;
  	ed[t]=n;
  	for(int i=1;i<=n;i++)
  		pos[i]=(i-1)/B+1;
  	init(n);
  	int op,x,y,z;
  	for(int i=1;i<=q;i++){
  		cin>>op;
  		if(op==1){cin>>x>>y>>z;change(x,y,z);}
  		if(op==2){cin>>x>>y;cout<<query(x,y)<<endl;}
  	}
  	return 0;
  }
  ```
