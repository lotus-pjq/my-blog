## P4570 [BJWC2011] 元素
### 与元素顺序有关的贪心法构造线性基例题
  - assets://D%3A/Logseq/assets/1973969-20240710204706550-1878120009_1755418439607_0.png
```C++
#include
using namespace std;
#define int long long
#define endl '\n'
const int N=1e4+5;
int n,p[64];
struct node{
int val,id;
bool operatorx.val;}
}a[N];
int ans=0;
void insert(int x,int pos){//贪心法
for(int i=63;i>=0;--i)
if(x>>i&1){
if(!p[i]){p[i]=x,ans+=a[pos].val;break;}//不存在则加入
x^=p[i];//存在则异或
}
}
signed main(){
ios::sync_with_stdio(false);
cin.tie(nullptr);
cin>>n;
for(int i=0;i>a[i].id>>a[i].val;}
sort(a,a+n);
for(int i=0;i<n;i++) insert(a[i].id,i);
cout<<ans;
return 0;
}
```
