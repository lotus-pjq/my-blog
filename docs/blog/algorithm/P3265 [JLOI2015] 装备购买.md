## P3265 [JLOI2015] 装备购买

#include

#define int long long

#define rep(i,j,k) for (int i=j;i=-eps) continue;
      if(!p[j]){
        p[j]=i;//记录最高位为第j位的线性基
        cnt++;ans+=q[i].w;
        break;
      }
      else{
        double d=q[i].a[j]/q[p[j]].a[j];
        rep(k,j,m) q[i].a[k]-=d*q[p[j]].a[k];
      }
    }
}
signed main(){
  ios::sync_with_stdio(false);
  cin.tie(nullptr);
  cin>>n>>m;
  rep(i,1,n)
    rep(j,1,m) cin>>q[i].a[j];
  rep(i,1,n) cin>>q[i].w;
  sort(q+1,q+n+1);//按花费升序
  guass();
  cout<<cnt<<" "<<ans;
  return 0;
}
