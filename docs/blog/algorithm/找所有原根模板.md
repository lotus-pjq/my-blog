- ```C++
  #include<bits/stdc++.h>
  using namespace std;
  #define int long long
  #define endl '\n'
  int pnum;
  vector<int> pri;
  vector<int> mpf,phi;//min prime pri_fac
  vector<int> rt;//是否存在原根
  void get_prime(int n){
  	mpf.resize(n+1,0);
  	phi.resize(n+1,0);phi[1]=1;
  	rt.resize(n+1,0);
  	for(int i=2;i<=n;i++){
  		if(mpf[i]==0){
  			mpf[i]=i,phi[i]=i-1,pnum++,pri.push_back(i);}
  		for(int p:pri){
  			if(i*p>n) break;
  			mpf[i*p]=p;
  			if(mpf[i]==p){phi[p*i]=phi[i]*p;break;}
  			else phi[i*p]=phi[i]*(p-1);
  		}
  	}
  	rt[2]=rt[4]=1;
  	for(int p:pri){
  		for(int j=1;(1ll*j*p)<=n;j*=p) rt[j*p]=1;
  		for(int j=2;(1ll*j*p)<=n;j*=p) rt[j*p]=1; 
  	}
  }
  int gcd(int a,int b){
  	int tmp;
  	while(b!=0){tmp=a%b;a=b;b=tmp;}
  	return a;
  }
  int ksm(int a,int b,int c){
  	int ans=1;
  	while(b){
  	   if(b&1) ans=(ans*a)%c;
  	   b>>=1;a=(a*a)%c;
  	}return ans;
  }
  vector<int> pri_fac;
  //将phi(n)的质因子存到pri_fac数组中
  void pre(int x){
  	pri_fac.clear();
  	int num=x,tmp;
  	while(num>1){
  		tmp=mpf[num];
  		pri_fac.push_back(tmp);
  		while(num>1&&num%tmp==0) num/=tmp;
  	}
  }
  bool check(int i,int x){
  	if(ksm(i,phi[x],x)!=1) return 0;
  	for(int pf:pri_fac)
  		if(ksm(i,phi[x]/pf,x)==1) return false;
  	return true;
  }
  int min_rt(int x){
  	for(int i=1;i<x;i++)
  		if(check(i,x)) return i;
  }
  vector<int> ans;
  void get_rt(int mn,int x){
  	int cur=1;
  	for(int i=1;i<=phi[x];i++){
  		cur=cur*mn%x;
  		if(gcd(i,phi[x])==1) ans.push_back(cur);
  	}
  }
  void solve(){
  	int n,d;cin>>n>>d;
  	ans.clear();
  	//原根数量
  	if(!rt[n]) cout<<0<<endl<<endl;
  	else{
  		cout<<phi[phi[n]]<<endl;
  		pre(phi[n]);
  		get_rt(min_rt(n),n);
  		sort(ans.begin(),ans.end());
  		for(auto c:ans) cout<<c<<" ";
  		cout<<endl;
  	} 
  }
  signed main(){
  	ios::sync_with_stdio(false);
  	cin.tie(nullptr);
  	get_prime(1000005);
  	int t;cin>>t;
  	while(t--) solve();
  	return 0;
  }
