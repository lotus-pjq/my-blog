- 原理：$[A|E]$在初等变换后等到$[E|A^{-1}]$
- 用高斯消元法使A被初等变换成E
- 若消元过程中出现了一行均为0，即矩阵不是满秩，故不存在逆元
- 因为是取模意义下，故需要费马小定理结合快速幂实现
- 代码：
  ```C++
  #include<bits/stdc++.h>
  using namespace std;
  #define int long long
  #define endl '\n'
  #define INF 0x3f3f3f3f3f3f3f3f
  #define rep(i,j,k) for (int i=j;i<=k;++i)
  #define per(i,j,k) for (int i=j;i>=k;--i)
  const int mod=1e9+7;
  const int N=4e2+5;
  int a[N][N<<1];
  int n;
  int ksm(int a,int b,int c=mod){
  	int ans=1;
  	while(b){
  	   if(b&1) ans=(ans*a)%c;
  	   b>>=1;a=(a*a)%c;}
  	return ans;
  }
  bool inv_matrix(){
  	for(int i=1;i<=n;++i){
  		int r=i;
  		for(int k=i;k<=n;++k)
  			if(a[k][i]){r=k;break;}
  		if(r!=i) swap(a[r],a[i]);
  		if(!a[i][i]) return 0;
  
  		int x=ksm(a[i][i],mod-2);
  		for(int k=1;k<=n;++k){
  			if(k==i) continue;
  			int t=a[k][i]*x%mod;
  			for(int j=i;j<=2*n;++j)
  				a[k][j]=((a[k][j]-t*a[i][j])%mod+mod)%mod;
  		}
  		for(int  j=1;j<=2*n;++j) 
  			a[i][j]=(a[i][j]*x%mod);		
  	}
  	return 1;
  }
  signed main(){
  	ios::sync_with_stdio(false);
  	cin.tie(nullptr);
  	cin>>n;
  	for(int i=1;i<=n;++i) 
  		for(int j=1;j<=n;++j)
  			cin>>a[i][j],a[i][i+n]=1;
  	if(inv_matrix())
  		for(int i=1;i<=n;++i){
  			for(int j=n+1;j<=2*n;j++)
  				cout<<a[i][j]<<" ";
  			cout<<endl;
  		}
  	else cout<<"No Solution";
  	return 0;
  }
  ```
