### 欧拉线性筛求[欧拉函数]、[莫比乌斯函数]、[最小质因数]、[约数个数]、[约数和] O(n)- ```C++
	  int pnum;
	  vector pri;
	  vector mpf,phi,mu;//mpf:min prime factor
	  vector g,f;//f[i]:i的约数和
	  //g[i]:i的最小质因子的1+p^1+...+p^k
	  vector b,d;//d[i]:i的约数个数
	  //b[i]:i的最小质因子的次数
	  void sieve(int n){
	  	mpf.resize(n+1,0);
	  	phi.resize(n+1,0);
	  	mu.resize(n+1,0);
	  	g.resize(n+1,0),f.resize(n+1,0);
	  	b.resize(n+1,0),d.resize(n+1,0);
	  	phi[1]=1;
	  	mu[1]=1;
	  	g[1]=f[1]=1;
	  	d[1]=1;
	  	for(int i=2;in) break;
	  			mpf[i*p]=p;
	  			if(mpf[i]==p){
	  				phi[p*i]=phi[i]*p;
	  				mu[p*i]=0;
	  				g[p*i]=g[i]*p+1;
	  				f[p*i]=f[i]/g[i]*g[p*i];
	  				b[p*i]=b[i]+1;
	  				d[p*i]=d[i]/b[p*i]*(b[p*i]+1);
	  				break;
	  			}
	  			else{
	  				phi[p*i]=phi[i]*(p-1);
	  				mu[p*i]=-mu[i];
	  				g[p*i]=p+1;
	  				f[p*i]=f[i]*g[p*i];
	  				b[p*i]=1;
	  				d[p*i]=d[i]*2;
	  			}
	  		}
	  	}
	  }
	  ```
