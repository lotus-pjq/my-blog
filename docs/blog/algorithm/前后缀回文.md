- ## Code
	- ```C++
	  #include<bits/stdc++.h>
	  using namespace std;
	  #define endl '\n'
	  #define int long long 
	  using ull=unsigned long long;
	  const int N=1e6+5;
	  const int P1=1341,M1=1e9+7;
	  const int P2=13331,M2=998244353;
	  int p1[N],h1[N],rh1[N];
	  int p2[N],h2[N],rh2[N];
	  #define pii pair<int,int>
	  string s;
	  //获取S[l...r]的正向哈希值
	  pii get(int l,int r){
	  	int len=r-l+1;
	  	int v1=(h1[r+1]-(h1[l]*p1[len])%M1+M1)%M1;
	  	int v2=(h2[r+1]-(h2[l]*p2[len])%M2+M2)%M2;
	  	return {v1,v2};
	  }
	  //获取RS[l...r]的哈希值
	  pii rget(int l,int r){
	  	int len=r-l+1;
	  	int v1=(rh1[r+1]-(rh1[l]*p1[len])%M1+M1)%M1;
	  	int v2=(rh2[r+1]-(rh2[l]*p2[len])%M2+M2)%M2;
	  	return {v1,v2};
	  }
	  //预处理S和其反转串RS的哈希值
	  void pre(int n){
	  	p1[0]=p2[0]=1;
	  	//p是基数幂表,用于快速截取子串哈希
	  	for(int i=1;i<=n;i++){
	  		p1[i]=(p1[i-1]*P1)%M1;
	  		p2[i]=(p2[i-1]*P2)%M2;
	  	}
	  	string rs=s;
	  	reverse(rs.begin(),rs.end());
	  	h1[0]=h2[0]=rh1[0]=rh2[0]=0;
	  	for(int i=0;i<n;++i){
	  		h1[i+1]=(h1[i]*P1+s[i])%M1;
	  		h2[i+1]=(h2[i]*P2+s[i])%M2;
	  		rh1[i+1]=(rh1[i]*P1+rs[i])%M1;
	  		rh2[i+1]=(rh2[i]*P2+rs[i])%M2;
	  	}
	  }
	  
	  void solve(){
	  	cin>>s;
	  	int n=s.length();
	  	if(n<=1){cout<<s<<endl;return;}
	  	int k=0;
	  	while(k<n/2&&s[k]==s[n-1-k]) k++;
	  	if(2*k>=n){cout<<s<<endl;return;}
	  	pre(n);
	  	int len=n-2*k;
	  	int b1=0,b2=0;
	  	//寻找中间部分的最长回文前缀
	  	for(int i=len;i>=1;--i){
	  		int l=k,r=k+i-1;
	  		if(get(l,r)==rget(n-1-r,n-1-l)){
	  			b1=i;break;
	  		}
	  	}
	  	//寻找中间部分的最长回文后缀
	  	for(int i=len;i>=1;--i){
	  		int l=n-k-i,r=n-1-k;
	  		if(get(l,r)==rget(n-1-r,n-1-l)){
	  			b2=i;break;
	  		}
	  	}
	  	cout<<s.substr(0,k);
	  	if(b1>=b2) cout<<s.substr(k,b1);
	  	else cout<<s.substr(n-k-b2,b2);
	  	cout<<s.substr(n-k)<<endl;
	  }
	  signed main(){
	  	ios::sync_with_stdio(false);
	  	cin.tie(nullptr);
	  	int t;cin>>t;
	  	while(t--) solve();
	  	return 0;
	  }
