## 感悟：
	- ### 1.用线段树解决问题，首先得考虑维护哪些信息。若不带修，任何 **满足结合律且封闭** 的信息（称为半群）都是可维护的。结合律一般都有，封闭性帮助我们设计信息。
	- ### 2.对于lazy tag的理解
		- ### lazy tag 的使用
		  其实这道题（维护历史最值）完全可以作为线段树的模板题，考察了 `lazy tag` 与树上节点的更新顺序的关系
		- ### 如何理解？
		  `lazy tag` 实际上可以看作是对于该节点表示的区间的**操作序列**，这也是线段树的精髓所在
		  `push_down` 操作就是把父节点的操作序列接在儿子节点的序列**之后**
		  （*思考：为什么一定是 “ **之后** ” ？*）
		  对一个点进行 `push_down` 操作后，该点的操作序列即被清空，因为在递归完子树后，该点答案将被更新。以下文章中的 “ 操作序列 ”，都指的是该点上一次进行 `push_down` 后（即上一次清空后）的操作序列
		- ### 如何处理操作序列？
		  首先，相邻的同种操作可被合并
		  于是，对于一个点，操作序列分为以下几种（ "+" 号表示操作的先后顺序，汉字叙述可能影响您的阅读体验）：
		- 加操作
		- 加操作+赋值操作
		- 加操作+赋值操作+加操作
		- 加操作+赋值操作+加操作+赋值操作
		- ⋯
		  这个操作序列显然很难保存在每一个点上
		  我们发现，对于一个点，首次赋值操作之后的任何修改都可以看作赋值操作（*想一想，为什么*）
		  于是操作序列化简为（**重点**）：
		  加操作 + 赋值操作
		  现在，恭喜您解决了两个个很大的问题：操作序列如何保存 & 同一点上操作的先后顺序问题
		  于是，只用保存加和赋值两种 `lazy tag`，便可记录该节点的操作序列
	- ### 3.区间修改需要lazy，单点修改不需要lazy。

	- 题目
		- 
	- 代码
		- ```C++
		  #include<bits/stdc++.h>
		  using namespace std;
		  #define int long long
		  #define ls x<<1
		  #define rs x<<1|1
		  const int N=5e5+5;
		  struct node{
		      int l,r;
		      int sum,lmx,rmx,mx;
		  }tr[N<<2];
		  int a[N];
		  void pushup(node&x,node l,node r){
		  	x.sum=l.sum+r.sum;
		  	x.lmx=max(l.lmx,l.sum+r.lmx);
		  	x.rmx=max(r.rmx,r.sum+l.rmx);
		  	x.mx=max(max(l.mx,r.mx),l.rmx+r.lmx);
		  }
		  void build(int x,int l,int r){
		  	tr[x]={l,r,a[l],a[l],a[l],a[l]};
		  	if(l==r) return;
		  	int mid=l+r>>1;
		  	build(ls,l,mid);
		  	build(rs,mid+1,r);
		  	pushup(tr[x],tr[ls],tr[rs]);
		  }
		  void change(int x,int pos,int val){
		      if(tr[x].l==tr[x].r){tr[x]={pos,pos,val,val,val,val};return;}
		      int mid=tr[x].l+tr[x].r>>1;
		      if(pos<=mid) change(ls,pos,val);
		      else change(rs,pos,val);
		      pushup(tr[x],tr[ls],tr[rs]);
		  }
		  node query(int x,int l,int r){
		      if(l<=tr[x].l&&tr[x].r<=r) return tr[x];
		      int mid=tr[x].l+tr[x].r>>1;
		      if(r<=mid) return query(ls,l,r);
		      if(l>=mid+1) return query(rs,l,r);
		      node tmp;
		      pushup(tmp,query(ls,l,mid),query(rs,mid+1,r));
		      return tmp;
		  }
		  int n,m;
		  signed main(){
		      ios::sync_with_stdio(false);
		  	cin>>n>>m;
		  	for(int i=1;i<=n;i++) cin>>a[i];
		  	build(1,1,n);
		  	while(m--){
		  		int op,x,y;cin>>op>>x>>y;
		  		if(op==1){
		  			if(x>y) swap(x,y);
		  			cout<<query(1,x,y).mx<<endl;
		  		}else change(1,x,y);
		  	}
		      cin.tie(0);
		      
		  }

		  #define int long long
		  #define endl '\n'

		  const int N=2e5+5;
		  const int INF=0x3f3f3f3f3f3f3f3f;
		  int m,p;
		  struct node{
		  	int l,r,mx;
		  }t[N<<2];
		  void pushup(int x){t[x].mx=max(t[ls].mx,t[rs].mx);}
		  void build(int x,int l,int r){
		  	t[x]={l,r,-INF};
		  	if(l==r) return;
		  	int mid=l+r>>1;
		  	build(ls,l,mid); build(rs,mid+1,r);
		  }
		  void change(int x,int pos,int val){//点修
		  	if(t[x].l==pos&&t[x].r==pos){t[x].mx=val;return;}
		  	int mid=t[x].l+t[x].r>>1;
		  	if(pos<=mid) change(ls,pos,val);
		  	if(pos>=mid+1) change(rs,pos,val);
		  	pushup(x);
		  }
		  int query(int x,int L,int R){//区查
		  	if(L<=t[x].l&&R>=t[x].r) return t[x].mx;
		  	int mid=t[x].l+t[x].r>>1;
		  	int res=-INF;
		  	if(L<=mid) res=query(ls,L,R);
		  	if(R>mid) res=max(res,query(rs,L,R));
		  	return res;
		  }
		  signed main(){
		  	ios::sync_with_stdio(false);
		  	cin.tie(nullptr);
		  	cin>>m>>p;
		  	build(1,1,m);
		  	char op;
		  	int x,n=0,tmp=0;
		  	while(m--){
		  		cin>>op>>x;
		  		if(op=='A') change(1,++n,(x+tmp)%p);
		  		else{
		  			tmp=query(1,n-x+1,n);
		  			cout<<tmp<<endl;
		  		}
		  	}
		  	return 0;
		  }

- ## 二分答案+线段树维护01区间
	- 题目(2016-2017 National Taiwan University World Final Team Selection Contest --- A)
		- 
	- 题意描述
		- 给定一个长度为n的序列a, 有m次操作，每次操作选择一个区间[L,R], 将区间内的数从小到大排序，或从大到小排序，问最终序列中间的数是什么。
	- 思路
		- 二分答案mid，将小于mid的数变成−1，大于mid的数变成1，等于mid的数为0，每次操作相当于询问某个区间内−1或1的个数，然后将区间重新染色，这个可以用线段树维护。最终如果中间的数是0，则二分的值就是答案，如果是−1说明答案偏大，否则答案偏小。
		- 时间复杂度: O(n\*logn\*logn)
	- 代码
		- ```C++
		  #include<bits/stdc++.h>
		  using namespace std;
		  #define int long long
		  #define endl '\n'
		  #define rep(i,j,k) for (int i=j;i<=k;++i)
		  #define ls x<<1
		  #define rs x<<1|1
		  const int N=1e5+5;
		  struct node{
		  	int l,r,lazy;
		  	int neg,zero,pos;
		  }t[N<<2];
		  int n,k,jud;
		  int a[N],ql[N],qr[N];
		  void up(int x){
		  	t[x].neg=t[ls].neg+t[rs].neg;
		  	t[x].zero=t[ls].zero+t[rs].zero;
		  	t[x].pos=t[ls].pos+t[rs].pos;
		  }
		  void down(int x){
		  	if(t[x].lazy!=-2){
		  		if(t[x].lazy==-1){
		  			t[ls].neg=t[ls].r-t[ls].l+1;
		  			t[ls].pos=t[ls].zero=0;
		  			t[rs].neg=t[rs].r-t[rs].l+1;
		  			t[rs].pos=t[rs].zero=0;
		  		}else if(t[x].lazy==0){
		  			t[ls].neg=t[ls].pos=t[rs].neg=t[rs].pos=0;
		  			t[ls].zero=t[rs].zero=1;
		  		}else if(t[x].lazy==1){
		  			t[ls].pos=t[ls].r-t[ls].l+1;
		  			t[rs].pos=t[rs].r-t[rs].l+1;
		  			t[ls].neg=t[ls].zero=t[rs].neg=t[rs].zero=0;
		  		}
		  		t[ls].lazy=t[rs].lazy=t[x].lazy;
		  		t[x].lazy=-2;
		  	}
		  }
		  void build(int x,int l,int r){
		  	t[x]={l,r,-2,0,0,0};
		  	if(l==r){
		  		if(a[l]<jud) t[x].neg=1;
		  		else if(a[l]==jud) t[x].zero=1;
		  		else t[x].pos=1;
		  		return;
		  	}
		  	build(ls,l,l+r>>1),build(rs,(l+r>>1)+1,r);
		  	up(x);
		  }
		  void update(int x,int l,int r,int val){
		  	if(t[x].l>r||t[x].r<l) return;
		  	if(t[x].l>=l&&t[x].r<=r){
		  		t[x].neg=t[x].zero=t[x].pos=0;
		  		if(val==-1) t[x].neg=t[x].r-t[x].l+1;
		  		else if(val==1) t[x].pos=t[x].r-t[x].l+1;
		  		else if(val==0) t[x].zero=1;
		  		t[x].lazy=val;
		  		return;
		  	}
		  	down(x);
		  	update(ls,l,r,val),update(rs,l,r,val);
		  	up(x);
		  }
		  void query(int x,int l,int r,int &neg,int &zero,int &pos){
		  	if(t[x].l>r||t[x].r<l){neg=zero=pos=0;return;}
		  	if(t[x].l>=l&&t[x].r<=r){neg=t[x].neg,zero=t[x].zero,pos=t[x].pos;return;}
		  	down(x);
		  	int n1,z1,p1,n2,z2,p2;
		  	query(ls,l,r,n1,z1,p1);
		  	query(rs,l,r,n2,z2,p2);
		  	neg=n1+n2,zero=z1+z2,pos=p1+p2;
		  }
		  signed main(){
		      ios::sync_with_stdio(false);
		      cin.tie(nullptr);
		  	cin>>n>>k;
		  	rep(i,1,n) cin>>a[i];
		  	rep(i,1,k) cin>>ql[i]>>qr[i];
		  	int l=1,r=n,mid,ans;
		  	while(l<=r){
		  		jud=mid=l+r>>1;
		  		build(1,1,n);
		  		int cnt_neg,cnt_zero,cnt_pos;
		  		for(int i=1,l,r;i<=k;i++){
		  			l=ql[i],r=qr[i];
		  			bool u=l<r;
		  			if(!u) swap(l,r);
		  			query(1,l,r,cnt_neg,cnt_zero,cnt_pos);
		  			if(!u){
		  				if(cnt_pos) update(1,l,l+cnt_pos-1,1);
		  				if(cnt_zero) update(1,l+cnt_pos,l+cnt_pos,0);
		  				if(cnt_neg) update(1,r-cnt_neg+1,r,-1);
		  			}else{
		  				if(cnt_neg) update(1,l,l+cnt_neg-1,-1);
		  				if(cnt_zero) update(1,l+cnt_neg,l+cnt_neg,0);
		  				if(cnt_pos) update(1,r-cnt_pos+1,r,1);
		  			}
		  		}
		  		query(1,(n+1)/2,(n+1)/2,cnt_neg,cnt_zero,cnt_pos);
		  		if(cnt_zero){ans=mid;break;}
		  		else if(cnt_neg==1) r=mid-1;
		  		else l=mid+1;
		  	}
		  	cout<<ans;
		      return 0;
		  }
		  ```
- ## C32 P1607 [USACO09FEB] Fair Shuttle G #线段树 #贪心

		- 
	- 分析
		- ==贪心：按区间右端点排序后枚举，结果更优（因为右端点靠前的区间对后续没有影响）==
		- 
		-
	- 代码（暴力）

		  //b[j]:经过位置j时车上的奶牛数

	- 代码（线段树）
		- ```C++
		  #include<bits/stdc++.h>
		  using namespace std;
		  #define int long long
		  #define endl '\n'
		  #define mid (l+r>>1)
		  #define ls x<<1
		  #define rs x<<1|1
		  const int N=2e5+5;
		  int n,k,c,ans;
		  struct cow{
		  	int l,r,m;
		  	bool operator<(const cow&t)const{return r<t.r;}
		  }s[N];
		  struct node{int l,r,mx,lazy;}t[N<<2];
		  void up(int x){t[x].mx=max(t[ls].mx,t[rs].mx);}
		  void down(int x){
		  	if(t[x].lazy){
		  		t[ls].mx+=t[x].lazy;
		  		t[rs].mx+=t[x].lazy;
		  		t[ls].lazy+=t[x].lazy;
		  		t[rs].lazy+=t[x].lazy;
		  		t[x].lazy=0;
		  	}
		  }
		  void build(int x,int l,int r){
		  	t[x]={l,r,0,0};
		  	if(l==r) return;
		  	build(ls,l,mid),build(rs,mid+1,r);
		  	up(x);
		  }
		  void change(int x,int l,int r,int v){
		  	if(l>t[x].r||r<t[x].l) return;
		  	if(l<=t[x].l&&t[x].r<=r){
		  		t[x].mx+=v,t[x].lazy+=v;
		  		return;
		  	}
		  	down(x);
		  	change(ls,l,r,v),change(rs,l,r,v);
		  	up(x);
		  }
		  int query(int x,int l,int r){
		  	if(l>t[x].r||r<t[x].l) return 0;
		  	if(l<=t[x].l&&t[x].r<=r) return t[x].mx;
		  	down(x);
