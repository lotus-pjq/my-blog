- 平面空间内两点，坐标为（x1,y1），（x2,y2）
- ## 曼哈顿距离：dis=|x1-x2|+|y1-y2|，即两点横纵坐标差之和
	- 用曼哈顿距离表示，则与原点距离为1的点会构成一个边长为√2的正方形
	- 
	-
- ## 切比雪夫距离：dis=max(|x1-x2|,|y1-y2|)，即两点横纵坐标差的最大值
	- 用切比雪夫距离表示，则与原点距离为1的点会构成一个边长为2的正方形
	- 
- ### Transform manhattan distance to chebyshev distance
- #### 变换：曼哈顿距离可以通过把坐标绕原点向左旋转45°，然后拉长√2，变换成切比雪夫距离。
- 同理可以给出坐标周的逆变换
- ---
- ## 坐标映射关系：
- 将一个点 (x,y) 的坐标变为( x+y , x−y )后,原坐标系中的曼哈顿距离 --> 新坐标系中的切比雪夫距离
- 将一个点 (x,y) 的坐标变为( (x+y)/2 ,(x-y)/2 ) 后,原坐标系中的切比雪夫距离 --> 新坐标系中的曼哈顿距离
- ---
- ## 用处
  切比雪夫距离在计算的时候需要取max，往往不是很好优化，对于一个点，计算其他点到该的距离的复杂度为O(n)
  而曼哈顿距离只有求和以及取绝对值两种运算，我们把坐标排序后可以去掉绝对值的影响，进而用前缀和优化，可以把复杂度降为O(1)

- ### P3964 [TJOI2013] 松鼠聚会
- (https://www.luogu.com.cn/problem/P3964)
	- ---
	- ### 题目描述
	  草原上住着一群小松鼠，每个小松鼠都有一个家。时间长了，大家觉得应该聚一聚。但是草原非常大，松鼠们都很头疼应该在谁家聚会才最合理。
	  每个小松鼠的家可以用一个点 $(x,y)$ 表示,两个点的距离定义为点 $(x,y)$ 和它周围的 $8$ 个点 $(x-1,y)$，$(x+1,y)$，$(x,y-1)$，$(x,y+1)$，$(x-1,y+1)$，$(x-1,y-1)$，$(x+1,y+1)$，$(x+1,y-1)$ 距离为 $1$。
	- ### 输入格式
	  第一行是一个整数 $N$，表示有多少只松鼠。接下来 $N$ 行，第 $i$ 行是两个整数 $x$ 和 $y$，表示松鼠 $i$ 的家的坐标。
	- ### 输出格式
	  一个整数，表示松鼠为了聚会走的路程和最小是多少。
	- 输入1
	  ```
	  6
	  -4 -1
	  -1 -2
	  2 -4
	  0 2
	  0 3
	  5 -2
	  ```
	- 输出1
	  
	  ```
	  20
	  ```
	- 输入2
	  
	  ```
	  6
	  0 0
	  2 0
	  -5 -2
	  2 -2
	  -1 2
	  4 0
	  ```
	- 输出2
	  
	  ```
	  15
	  ```
	- ### 样例解释
	  在第一个样例中，松鼠在第二只松鼠家 $(-1,-2)$ 聚会；在第二个样例中，松鼠在第一只松鼠家 $(0,0)$ 聚会。
	- ### 数据范围
	- $30\%$ 的数据，$0\le N \le 1000$；
	- $100\%$ 的数据，$0\le N \le 10^5$，$-10^9 \le x, y \le 10^9$。

	  const int N=2e5+5;
	  int x[N],y[N],u[N],v[N],prex[N],prey[N];

	  		cin>>xx>>yy;
	  		u[i]=x[i]=xx+yy,v[i]=y[i]=xx-yy;
	  	}
	  	sort(u+1,u+n+1);sort(v+1,v+n+1);
	  	for(int i=1;i<=n;i++) 
	  		prex[i]=prex[i-1]+u[i],
	  		prey[i]=prey[i-1]+v[i];
	  	int ans=INF,sumx,sumy,posx,posy;
	  	for(int i=1;i<=n;i++){
	  		posx=lower_bound(u+1,u+n+1,x[i])-u;
	  		posy=lower_bound(v+1,v+n+1,y[i])-v;
	  		sumx=posx*u[posx]-prex[posx]+(prex[n]-prex[posx])-(n-posx)*u[posx];
	  		sumy=posy*v[posy]-prey[posy]+(prey[n]-prey[posy])-(n-posy)*v[posy];
	  		ans=min(ans,(sumx+sumy)/2);
	  	}
	  	cout<<ans;
	  	return 0;
	  }

- 牛客多校5.J
	- 
	- 二分答案考虑答案为k时，判断其是否有解；
	- 可以先BFS求出所有点到1的最短距离，然后可以快速得到所有还未被覆盖的点，即所有大于mid的点形成的集合,用一个矩阵包围它,那就是判断以2*mid的正方形（斜着）能不能覆盖整个矩形。
	- 把曼哈顿距离转化成切比雪夫距离后这个线性规划的可行域和正方形都是正着的，只需要判断上下和左右约束的直线是不是都小于等于mid即可判断mid是否可行。
	- ```C++
	  #include<bits/stdc++.h>
	  using namespace std;
	  #define int long long
	  #define endl '\n'
	  #define INF 0x3f3f3f3f3f3f3f3f
	  const int dir[4][2]={ {0,1},{1,0},{0,-1},{-1,0} };
	  signed main(){
	  	ios::sync_with_stdio(false);
	  	cin.tie(nullptr);
	  	int n,m;cin>>n>>m;
	  	vector<vector<int>> mat(n+1,vector<int>(m+1));
	  	vector<vector<int>> dis(n+1,vector<int>(m+1,INF));
	  	queue<pair<int,int>> que;
	  	bool f=false;
	  	for(int i=1;i<=n;i++)
	  		for(int j=1;j<=m;j++){
	  			cin>>mat[i][j];
	  			if(mat[i][j]) f=1,que.emplace(i,j),dis[i][j]=0;
	  		}
	  	if(!f){cout<<n/2+m/2;return 0;}
	  	while(!que.empty()){
	  		auto [x,y]=que.front();
	  		que.pop();
	  		for(int d=0;d<4;d++){
	  			int xx=x+dir[d][0],yy=y+dir[d][1];
	  			if(xx>=1&&xx<=n&&yy>=1&&yy<=m&&dis[xx][yy]==INF)
	  				dis[xx][yy]=dis[x][y]+1,que.emplace(xx,yy);
	  		}
	  	}
	  	int l=0,r=n+m-2,mid;
	  	while(l<=r){
	  		mid=l+r>>1;
	  		int w,s,a,d;//转换成切比雪夫距离后的上下左右四个方向
	  		w=d=-INF,a=s=INF;
	  		for(int i=1;i<=n;i++)
	  			for(int j=1;j<=m;j++)
	  				if(dis[i][j]>mid){
	  					int x=i+j,y=i-j;
	  					a=min(a,x),w=max(w,y),d=max(d,x),s=min(s,y);
	  				}
	  		if(d-a<=2*mid&&w-s<=2*mid) r=mid-1;
	  		else l=mid+1;
	  	}
	  	cout<<l<<endl;
	  	return 0;
	  }
	  ```

	  ```C++
	  #include<bits/stdc++.h>
	  using namespace std;
	  #define INF 0x3f3f3f3f3f3f3f3f
	  #define int long long
	  #define endl '\n'
	  const int mod=1e9+7;
	  const int inv2=(mod+1)>>1;
	  const int N=2e5+5;
	  int n,m,a[N],suma;
	  void add(int &a,int b){b%=mod;a+=b;a+=mod,a%=mod;}
	  int cal(vector<pair<int,int>>&v){
	  	//(i<j) ai*wj + aj*wj - ai*wi -aj*wi
	  	int prea=0,pt1=0,pt2=0,pt3=0,pt4=0,p=0;
	  	for(auto [w,a]:v){
	  		p++;
	  		add(pt4,w*(suma-prea+mod));
	  		add(prea,a);
	  		add(pt1,w*prea);
	  		add(pt2,a*w%mod*p);
	  		add(pt3,a*w%mod*(n+1-p));
	  	}
	  	int ret=(pt1+pt2-pt3-pt4+2*mod)%mod;
	  	return ret;
	  }
	  void solve(){
	  	cin>>n;
	  	suma=0;
	  	vector<pair<int,int>>rA,rB;
	  	for(int i=1,x,y,a;i<=n;i++){
	  		cin>>x>>y>>a;
	  		add(suma,a);
	  		rA.push_back({x+y,a});
	  		rB.push_back({x-y,a});
	  	}
	  	sort(rA.begin(),rA.end());
	  	sort(rB.begin(),rB.end());
	  	int ans=(cal(rA)+cal(rB))%mod*inv2%mod;
	  	cout<<ans<<endl;
	  }
	  signed main(){
	  	cin.tie(nullptr);
	  	int t;cin>>t;
	  	while(t--) solve();
	  	return 0;
	  }
