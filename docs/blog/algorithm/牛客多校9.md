		  using namespace std;

		  const int N=2e5+5;
		  const int M=30;
		  int n,l[N],r[N];
		  int dp[N][M+5];
		  int f[M+5]; 
		  void init(){
		      f[0]=0; f[1]=1; f[2]=2;
		      rep(i,0,M-1) dp[0][i] = INF;
		      dp[0][0]=0; dp[0][1]=1; dp[0][2]=2;
		      rep(i,3,M-1) {
		          f[i]=f[i-1]+f[i-2]+1;
		          dp[0][i]=f[i];//空树的高度为i所需删除次数
		      }
		  }
		  void dfs(int x){
		      rep(i,0,M) dp[x][i]=INF; 
		      if(l[x]) dfs(l[x]);
		      if(r[x]) dfs(r[x]);
		      dp[x][0] = dp[l[x]][0] + dp[r[x]][0] + 1;
		      rep(i,1,M){ 
		          int s=INF;
		          s = min(s, dp[l[x]][i-1] + dp[r[x]][max(i-2,0LL)]);
		          s = min(s, dp[l[x]][i-1] + dp[r[x]][i-1]);
		          s = min(s, dp[r[x]][i-1] + dp[l[x]][max(i-2,0LL)]);
		          dp[x][i] = s;
		      }
		  }
		  void solve(){
		      cin>>n;
		      rep(i,1,n) cin>>l[i]>>r[i];
		      int ans=INF;
		      dfs(1);
		      rep(i,1,M) ans=min(ans,dp[1][i]);
		      cout<<ans<<endl;
		  }
		  signed main(){
		      ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);
		      init();
		      int t=1; cin>>t;
		      while(t--) solve();
		  }

		- ```C++
		  #include<bits/stdc++.h>
		  using namespace std;
		  #define int long long
		  #define endl '\n'
		  #define INF 0x3f3f3f3f3f3f3f3f
		  #define rep(i,j,k) for (int i=j;i<=k;++i)
		  #define per(i,j,k) for (int i=j;i>=k;--i)
		  #define mid (l+r>>1)
		  const int N=1e6+5;
		  const int mod=998244353;
		  int n,a[N];
		  struct node{
		  	int mn,pos;
		  }t[N<<2];
		  void up(int x){
		  	if(t[x<<1].mn<t[x<<1|1].mn) t[x]=t[x<<1];
		  	else t[x]=t[x<<1|1];
		  }
		  void build(int x,int l,int r){
		  	if(l==r){t[x]={a[l],l};return;}
		  	build(x<<1,l,mid);
		  	build(x<<1|1,mid+1,r);
		      up(x);
		  }
		  int query(int x,int l,int r,int L,int R){
		  	if(L<=l&&r<=R) return t[x].pos;
		  	int posl=0,posr=0;
		  	if(L<=mid) posl=query(x<<1,l,mid,L,R);
		  	if(R>mid) posr=query(x<<1|1,mid+1,r,L,R);
		  	if(a[posl]<a[posr]) return posl;
		  	else return posr;
		  }
		  int tt[N<<1],intt[N<<1];
		  void init_inff(){
		  	tt[0]=intt[0]=1;
		  	tt[1]=intt[1]=1;
		  	for(int i=2;i<2*N;i++){
		  		tt[i]=i;
		  		intt[i]=(mod-mod/i)*intt[mod%i]%mod;
		  	}
		  	for(int i=2;i<2*N;i++){
		  		tt[i]=tt[i-1]*tt[i]%mod;
		  		intt[i]=intt[i-1]*intt[i]%mod;
		  	}
		  }
		  int C(int n,int m){
		  	if(n<0||m<0||m>n) return 0;
		  	return tt[n]*intt[m]%mod*intt[n-m]%mod;
		  }
		  int ans;
		  int dfs(int l,int r,int s){//s:当前节点上方节点数量
		      if(r<l)return 0;
		      int w=query(1,1,n,l,r);
		      int d=1;//子树大小
		      d+=dfs(l,w-1,s+1);
		      d+=dfs(w+1,r,s+1);
		      ans=(ans+C(d+s,s+1))%mod;
		      return d;
		  }
		  void solve(){
		      cin>>n;int i;ans=0;
		      for(i=1;i<=n;i++) cin>>a[i];
		  	build(1,1,n);
		      dfs(1,n,0);
		      cout<<(ans+1)%mod<<endl;
		  }
		  signed main(){
		      ios::sync_with_stdio(0);
		      cin.tie(0);cout.tie(0);
		  	init_inff(); a[0]=INF;
		      int t=1;cin>>t;
		      while(t--){solve();}
		  }
