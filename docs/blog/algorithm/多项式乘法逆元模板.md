## 🧩 1. 题目抽象
给定多项式  
F(x) = f₀ + f₁x + f₂x² + … + fₙ₋₁xⁿ⁻¹  (f₀ ≠ 0)  
求多项式 G(x) 使得  
F(x) · G(x) ≡ 1 (mod xⁿ)  
即 G(x) 是 F(x) 在模 xⁿ 意义下的 **逆元**。  
输出 G(x) 的前 n 个系数（低次 → 高次）。
---
- ## ⚙️ 2. 算法总览
  使用 **牛顿迭代** + **NTT**（数论变换）
- 牛顿迭代：把求逆元转化为若干次 **多项式乘法**
- NTT：在 O(n log n) 时间内完成乘法
  ---
- ## 📦 3. 全局变量
  | 名字 | 含义 |
  |---|---|
  | `f[N]` | 原多项式 F(x) 的系数 |
  | `g[N]` | 当前迭代得到的逆元 G(x) |
  | `tmp[N]` | 临时数组，存乘法结果 |
  | `R[N]` | NTT 用到的蝴蝶变换（bit-reversal）下标 |
  | `g=3` | 模数 998244353 的原根 |
  | `mod=998244353` | NTT 的模数 |

- ```C++
  #include<bits/stdc++.h>
  using namespace std;
  #define int long long
  #define endl '\n'
  #define INF 0x3f3f3f3f3f3f3f3f
  #define rep(i,j,k) for (int i=j;i<=k;++i)
  #define per(i,j,k) for (int i=j;i>=k;--i)
  #define fi first
  #define se second
  #define clr(f,n) memset(f,0,sizeof(int)*((n)+1))
  #define cpy(f,g,n) memcpy(f,g,sizeof(int)*((n)+1))
  const double eps=1e-12;
  const int mod=998244353;       // 多项式模数
  const double PI=acos(-1);
  /* 题目描述：给定一个 n 次多项式 F(x)，保证常数项不为 0
     请求出 G(x) 满足 F(x)·G(x) ≡ 1 (mod x^n)
     即输出 G(x) 的前 n 项系数（从低到高）
     输入格式：第一行一个整数 n
               接下来一行 n 个整数，从低到高表示 F(x) 的系数
     输出格式：一行 n 个整数，从低到高表示 F(x) 的逆元系数
  */
  const int N=4e6+5;  //多项式最大长度
  const int g=3;  //模数 998244353 的原根
  int n,F[N];     //原多项式 F(x) 的系数
  int G[N];       //当前迭代得到的逆元 G(x)
  int R[N];       //NTT用到的蝴蝶变换（bit-reversal）下标
  int tmp[N];     //临时数组，存乘法结果
  int ksm(int a,int b,int p=mod){
      int res=1;
      while(b){
          if(b&1) res=res*a%p;
          a=a*a%p; b>>=1;
      }   return res;
  }
  void NTT(int *A,int len,int op){
      // step1: 蝴蝶变换
      for(int i=0;i<len;i++)
          R[i]=(R[i>>1]>>1)|((i&1)?len>>1:0);
      for(int i=0;i<len;i++)
          if(i<R[i]) swap(A[i],A[R[i]]);
      // step2: 分治 NTT
      for(int i=2;i<=len;i<<=1){
          int gn=ksm(g,(mod-1)/i);          // 本原单位根
          if(op==-1) gn=ksm(gn,mod-2);      // 逆变换取逆元
          for(int j=0;j<len;j+=i){
              int gk=1;
              for(int k=j;k<j+i/2;k++){
                  int x=A[k], y=1LL*gk*A[k+i/2]%mod;
                  A[k]=(x+y)%mod;
                  A[k+i/2]=(x-y+mod)%mod;
                  gk=1LL*gk*gn%mod;
              }
          }
      }
      // step3: 逆变换需除以 len
      if(op==-1){
          int inv=ksm(len,mod-2);
          for(int i=0;i<len;i++) A[i]=1LL*A[i]*inv%mod;
      }
  }
  void polyInv(int *f,int *g,int n){
      int lim=1; while(lim<=n) lim<<=1;   // 扩展到 2 的幂
      clr(g,lim);                        // 清空结果
      g[0]=ksm(f[0],mod-2);              // 逆元常数项
      for(int len=2;len<=lim;len<<=1){   // 倍增长度
          int m=len<<1;                  // 两倍长度做卷积
          // 1) 把 f 的低 len 项拷到 tmp，其余补 0
          clr(tmp,m);
          cpy(tmp,f,len);
          // 2) 三次 NTT：g, tmp 正变换 → 点乘 → 逆变换
          NTT(g,m,1); NTT(tmp,m,1);
          for(int i=0;i<m;i++)
              g[i]=(2LL-1LL*g[i]*tmp[i]%mod+mod)%mod*g[i]%mod;
          NTT(g,m,-1);
          // 3) 截断，只保留低 len 项
          clr(g+len,len);
      }
  }
  signed main(){
      ios::sync_with_stdio(false);
      cin.tie(nullptr);
      cin>>n;
  	n-=1;
      rep(i,0,n) cin>>F[i];   // 读入 F(x) 的系数
      polyInv(F,G,n);           // 计算逆元
      rep(i,0,n) cout<<G[i]<<" ";
      cout<<endl;
      return 0;
  }
  ```

- ## 🔧 4. 工具函数
- ### 4.1 `ksm(a,b)` —— 快速幂
  ```cpp
  int ksm(int a,int b,int p=mod){
      int res=1;
      while(b){
          if(b&1) res=res*a%p;
          a=a*a%p; b>>=1;
      }   return res;
  }
  ```
- 计算 aᵇ mod p
  ---
- ### 4.2 `NTT(A,len,op)` —— 数论变换
  ```cpp
  void NTT(int *A,int len,int op){
      // step1: 蝴蝶变换
      for(int i=0;i<len;i++)
          R[i]=(R[i>>1]>>1)|((i&1)?len>>1:0);
      for(int i=0;i<len;i++)
          if(i<R[i]) swap(A[i],A[R[i]]);
      // step2: 分治 NTT
      for(int i=2;i<=len;i<<=1){
          int gn=ksm(g,(mod-1)/i);          // 本原单位根
          if(op==-1) gn=ksm(gn,mod-2);      // 逆变换取逆元
          for(int j=0;j<len;j+=i){
              int gk=1;
              for(int k=j;k<j+i/2;k++){
                  int x=A[k], y=1LL*gk*A[k+i/2]%mod;
                  A[k]=(x+y)%mod;
                  A[k+i/2]=(x-y+mod)%mod;
                  gk=1LL*gk*gn%mod;
              }
          }
      }
      // step3: 逆变换需除以 len
      if(op==-1){
          int inv=ksm(len,mod-2);
          for(int i=0;i<len;i++) A[i]=1LL*A[i]*inv%mod;
      }
  }
  ```
- `op=1` 正变换，`op=-1` 逆变换
- 复杂度 O(len*log(len))即O(NlogN)

- ## 🔄 5. 多项式逆元主体 `polyInv(f,g,n)`
- ### 5.1 牛顿迭代公式
  设 Gₖ(x) 是 F(x) 在模 x²ᵏ 意义下的逆元，则  
  Gₖ₊₁(x) ≡ Gₖ(x) · (2 − F(x)·Gₖ(x)) (mod x²ᵏ⁺¹)
- ### 5.2 代码流程
  ```cpp
  void polyInv(int *f,int *g,int n){
      int lim=1; while(lim<n) lim<<=1;   // 扩展到 2 的幂
      clr(g,lim);                        // 清空结果
      g[0]=ksm(f[0],mod-2);              // 逆元常数项
      for(int len=2;len<=lim;len<<=1){   // 倍增长度
          int m=len<<1;                  // 两倍长度做卷积
          // 1) 把 f 的低 len 项拷到 tmp，其余补 0
          clr(tmp,m);
          cpy(tmp,f,len);
          // 2) 三次 NTT：g, tmp 正变换 → 点乘 → 逆变换
          NTT(g,m,1); NTT(tmp,m,1);
          for(int i=0;i<m;i++)
              g[i]=(2LL-1LL*g[i]*tmp[i]%mod+mod)%mod*g[i]%mod;
          NTT(g,m,-1);
          // 3) 截断，只保留低 len 项
          clr(g+len,len);
      }
  }
  ```
- 每轮迭代长度翻倍，总复杂度  
  T(n) = T(n/2) + O(n log n) ⇒ O(n log n)
